235 /*
236  * This API gets the tunnel to bridge/VLAN mapping information.
237  *
238  * The function can receive three types of input:
239  *
240  * - 1) cmd SX_ACCESS_CMD_GET_FIRST,
241  * get a list of first n<=1000 map entries of specified tunnel id.
242  *
243  * - 2) cmd SX_ACCESS_CMD_GETNEXT,
244  * get a list of n<=1000 next map entries whose value larger
245  * than the given map_entry_key of specified tunnel id.
246  *
247  * Supported devices: Spectrum, Spectrum2.
248  *
249  * @param[in] handle - SX-API handle.
250  * @param[in] cmd - GET_FIRST/ GET_NEXT.
251  * @param[in] tunnel_id - Tunnel ID
252  * @param[in] map_entry_key - map entry, as a key to get next entries
253  * @param[out] map_entries_p - Array of map entries
254  * @param[in,out] map_entries_cnt_p - amount of entries in map_entries_p
255  *
256  * @return SX_STATUS_SUCCESS if operation completes successfully.
257  * @return SX_STATUS_CMD_UNSUPPORTED if access command isn't supported.
258  * @return SX_STATUS_PARAM_EXCEEDS_RANGE if parameters exceed range.
259  * @return SX_STATUS_PARAM_ERROR if any input parameter is invalid.
260  * @return SX_STATUS_ENTRY_NOT_FOUND if neighbor was not added.
261  * @return SX_STATUS_MODULE_UNINITIALIZED when tunnel module is uninitialized
262  * @return SX_STATUS_ERROR general error.
263  */

266  const sx_tunnel_id_t tunnel_id,
267  sx_tunnel_map_entry_t map_entry_key,

269  uint32_t * map_entries_cnt_p);
