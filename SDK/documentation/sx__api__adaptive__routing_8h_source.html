274 /*
275  * This API sets the AR shaper rate configuration parameters.
276  * To prevent moving a large amount of flows moving from one output
277  * to another output in a short period of time
278  * two types of shapers per port are used:
279  *
280  * A shaper per output port for the number of flows that changed their destination to the output port
281  * A shaper per output port for the number of flows that changed their destination from the output port
282  *
283  * Supported devices: Spectrum2, Spectrum3.
284  *
285  * @param[in] handle - SX-API handle.
286  * @param[in] cmd - SET.
287  * @param[in] shaper_attr_p - Shaper configuration attributes.
288  *
289  * @return SX_STATUS_SUCCESS if operation completes successfully.
290  * @return SX_STATUS_INVALID_HANDLE if a NULL handle is received.
291  * @return SX_STATUS_MODULE_UNINITIALIZED if AR module wasn't initialized.
292  * @return SX_STATUS_PARAM_ERROR if parameter there is a bad parameter.
293  * @return SX_STATUS_PARAM_NULL if parameter is NULL.
294  * @return SX_STATUS_CMD_UNSUPPORTED if access command isn't supported.
295  * @return SX_STATUS_ERROR general error.
296  */

299  const sx_ar_shaper_attr_t *shaper_attr_p);
